rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Profile photos - user can read all, write own only
    match /profile_photos/{userId}.jpg {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId
                   && request.resource.size < 5 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }

    // Chat media - authenticated users can read/write in their conversations
    match /media/{conversationId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && conversationId.matches('.*' + request.auth.uid + '.*')
                   && request.resource.size < 50 * 1024 * 1024;
    }

    // Group media
    match /group_media/{groupId}/{fileName} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.resource.size < 50 * 1024 * 1024;
    }
  }
}
